276 void Blueprint::consistencyCheck() const
277 {
278  //Check algorithm and boundaries
279  if( alg.dt <= 0)
280  throw Message( "dt <= 0!\n", _ping_);
281  if( fabs( alg.h - bound.lx/(double)alg.nx) > 1e-15)
282  throw Message( "h != lx/nx\n", _ping_);
283  if( fabs( alg.h - bound.ly/(double)alg.ny) > 1e-15)
284  throw Message( "h != ly/ny\n", _ping_);
285  if( alg.nx == 0||alg.ny == 0)
286  throw Message( "Set nx and ny!\n", _ping_);
287  //Check physical parameters
288  if( phys.nu < 0)
289  throw Message( "nu < 0!\n", _ping_);
290  if( phys.a[0] <= 0 || phys.mu[0] <= 0 || phys.tau[0] < 0)
291  throw Message( "Ion species badly set\n", _ping_);
292  if( imp && (phys.a[1] < 0 || phys.mu[1] <= 0 || phys.tau[1] < 0))
293  throw Message( "Impuritiy species badly set\n", _ping_);
294  if( fabs(phys.a[0] + phys.a[1] - 1) > 1e-15)
295  throw Message( "a_i + a_z != 1 (background not neutral)\n", _ping_);
296  if( fabs( phys.g_e - phys.a[0]*phys.g[0]- phys.a[1]*phys.g[1]) > 1e-15)
297  throw Message( "Background is not neutral! \n", _ping_);
298  //inconsistency when impurities are not set??
299  if( !imp && (phys.a[1] != 0 || phys.mu[1] != 0 || phys.tau[1] != 0))
300  throw Message( "Impurity disabled but z species not 0!\n", _ping_);
301  //Some Warnings
302  if( global && (phys.g_e != 0||phys.g[0] != 0||phys.g[1] != 0))
303  std::cerr << "TL_WARNING: Global solver ignores gradients\n";
304 
305 }
