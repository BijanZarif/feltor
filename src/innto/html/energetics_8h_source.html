137 {

149  out(rows-1,j) = (in(0,j) - in(rows-2, j))/2./h;

165  crows( rows), ccols( cols/2+1),
166  diff_coeff( rows, cols),

168  dens_( MatrixArray<double, TL_DFT,n>::construct( rows, cols)),
169  phi_( dens_),
170  cdens_( MatrixArray<complex, TL_NONE, n>::construct( crows, ccols)),
171  cphi_(cdens_),
172  blue(bp), phys( bp.physical()), bound( bp.boundary()), alg( bp.algorithmic()),
173  dft_dft( rows, cols, FFTW_MEASURE),
174  a_{-1. , phys.a[0], phys.a[1]}, tau_{-1. , phys.tau[0], phys.tau[1]},
175  g_{phys.g_e, phys.g[0], phys.g[1]}
176  {
177  double laplace;
178  Poisson p( phys);
179  int ik;
180  const double kxmin2 = 2.*2.*M_PI*M_PI/(double)(bound.lx*bound.lx),
181  kymin2 = 2.*2.*M_PI*M_PI/(double)(bound.ly*bound.ly);
182  for( unsigned i = 0; i<crows; i++)
183  for( unsigned j = 0; j<ccols; j++)
184  {
185  ik = (i>rows/2) ? (i-rows) : i; //integer division rounded down
186  laplace = - kxmin2*(double)(j*j) - kymin2*(double)(ik*ik);
187  diff_coeff(i,j) = -phys.nu*pow(-laplace,2);
188  if( n==2)
189  {
190  a_mu_gamma0_coeff[0](i,j) = -phys.a[0]*phys.mu[0]*laplace*p.gamma0_i( laplace);
191  }
192  else if( n==3)
193  {
194  a_mu_gamma0_coeff[0](i,j) = -phys.a[0]*phys.mu[0]*laplace*p.gamma0_i( laplace);
195  a_mu_gamma0_coeff[1](i,j) = -phys.a[1]*phys.mu[1]*laplace*p.gamma0_z( laplace);
196  }
197  }

233 template<size_t n>
234 std::vector<double> Energetics<n>::exb_energies(const Matrix<double, TL_DFT>& potential )
235 {
236  std::vector<double> energies;
237  double norm = alg.h*alg.h/(double)(rows*cols); //h^2 from integration and 1/NxNy from complex scalar product
238  phi_[0] = potential;
239  dft_dft.r2c( phi_[0], cphi_[0]);
240  for( size_t k=1; k<n; k++)
241  {
242 #pragma omp parallel for
243  for( size_t i = 0; i < crows; i++)
244  for( size_t j = 0; j < ccols; j++)
245  cphi_[k](i,j) = (a_mu_gamma0_coeff[k-1](i,j))*cphi_[0](i,j);
246  energies.push_back( 0.5*dft_dft.dot( cphi_[0], cphi_[k])*norm);
247  }
248  //averages
249  double norm_avg = alg.h*alg.h/(double)(rows)/(double)cols; //h*h from integration 1/Ny from average and 1/Nx from complex scalar product
250  std::vector<complex> sum_phi[n];
251  for( unsigned i=0; i<n; i++)
252  sum_phi[i] = extract_sum_y( cphi_[i]);
253  for( size_t k=1; k<n; k++)
254  energies.push_back( 0.5*dot( sum_phi[0], sum_phi[k])*norm_avg);
255 
256  return energies;
257 }

271 double Energetics<n>::capital_a( const Matrix<double, TL_DFT>& density , const Matrix<double, TL_DFT>& potential)
272 {
273  std::vector<double> sum_dens = extract_sum_y( density);
274  std::vector<double> sum_phi = extract_sum_y( potential);
275  double a = dot( sum_phi, sum_phi) - dot(sum_dens, sum_phi);
276  return -phys.d*a/(double)(rows)*alg.h*alg.h;

340 template<size_t n>
341 std::vector<double> Energetics<n>::diffusion( const std::array<Matrix<double, TL_DFT>, n>& density , const std::array<Matrix<double, TL_DFT>, n>& potential)
342 {
343  //compute -nu*laplace^2 density in dens_
344  dens_ = density;
345 #pragma omp parallel for
346  for( unsigned k=0; k<n; k++)
347  {
348  dft_dft.r2c( dens_[k], cdens_[k]);
349  for( size_t i = 0; i < crows; i++)
350  for( size_t j = 0; j < ccols; j++)
351  cdens_[k](i,j) = diff_coeff(i,j)*cdens_[k](i,j)/(double)(rows*cols);
352  dft_dft.c2r( cdens_[k], dens_[k]);
353  }
354  //add up diffusion
355  std::vector<double> diffusion_(2, 0.);
356  for( unsigned i=0; i<n; i++)
357  {
358  diffusion_[0] += a_[i]*dot( potential[i], dens_[i])*alg.h*alg.h;
359  diffusion_[0] += a_[i]*tau_[i]*dot( density[i], dens_[i])*alg.h*alg.h;
360  }
361  //compute mean diffusion
362  for(unsigned k=0; k<n; k++)
363  {
364  std::vector<double> sum_dens = extract_sum_y( dens_[k] );
365  std::vector<double> sum_phi = extract_sum_y( potential[k] );
366  diffusion_[1] += a_[k]*dot( sum_dens, sum_phi)*alg.h*alg.h/(double)(rows);
367  }
