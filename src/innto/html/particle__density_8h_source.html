53  dft_dft.r2c( grad_phi[0], cgrad_phi[0]);
54  dft_dft.r2c( grad_phi[1], cgrad_phi[1]);
55  const complex dymin( 0, 2.*M_PI/bound.ly);
56  const complex dxmin( 0, 2.*M_PI/bound.lx);
57  double norm = 1./(double)(rows*cols);
58  complex dx, dy;
59  int ik;
60  for( unsigned i=0; i<crows; i++)
61  for( unsigned j=0; j<ccols; j++)
62  {
63  ik = (i>rows/2) ? (i-rows) : i; //integer division rounded down
64  if( rows%2 == 0 && i == rows/2) ik = 0;
65  dx = (double)j*dxmin;

68  cgrad_phi[1]( i, j) *= dy*norm;
