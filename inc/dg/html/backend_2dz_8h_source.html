143  dz_(field, grid.global(), deltaPhi, eps, limit, globalbcz)
144 {
145  assert( deltaPhi == grid.hz() || grid.Nz() == 1);
146  //2D local grid with ghostcells
147  dg::Grid2d<double> g2d( g_.x0(), g_.x1(), g_.y0(), g_.y1(), g_.n(), g_.Nx(), g_.Ny());
148  dg::Grid3d<double> global( grid.global()); //global without ghostcells
149  dg::Grid3d<double> globalWG( global.x0() - global.hx()*(1-1e-14), global.x1() + global.hx()*(1-1e-14),
150  global.y0() - global.hy()*(1-1e-14), global.y1() + global.hy()*(1-1e-14),
151  global.z0(), global.z1(),
152  global.n(), global.Nx()+2, global.Ny()+2, global.Nz());
153  //global with ghost-boundary for boxintegrator
154  limiter_ = dg::evaluate( limit, g2d);
155  right_ = left_ = dg::evaluate( zero, g2d);
156  ghostM.resize( g2d.size()); ghostP.resize( g2d.size());
157  //set up grid points as start for fieldline integrations (but not the ghostcells)
158  std::vector<dg::HVec> y( 3);
159  y[0] = dg::evaluate( dg::coo1, grid.local());
160  y[1] = dg::evaluate( dg::coo2, grid.local());
161  y[2] = dg::evaluate( dg::zero, grid.local());//distance (not angle)
162  //integrate to next z-planes
163  std::vector<dg::HVec> yp(y), ym(y);
164  thrust::host_vector<double> coords(3), coordsP(3), coordsM(3);
165  for( unsigned i=0; i<grid.size(); i++)
166  {
167  coords[0] = y[0][i], coords[1] = y[1][i], coords[2] = y[2][i];
168  double phi1 = deltaPhi;
169  boxintegrator( field, globalWG, coords, coordsP, phi1, eps, globalbcz);
170  phi1 = -deltaPhi;
171  boxintegrator( field, globalWG, coords, coordsM, phi1, eps, globalbcz);
172  yp[0][i] = coordsP[0], yp[1][i] = coordsP[1], yp[2][i] = coordsP[2];
173  ym[0][i] = coordsM[0], ym[1][i] = coordsM[1], ym[2][i] = coordsM[2];
174  }
175 
176 
177  //determine pid of result
178  thrust::host_vector<int> pids( grid.size());
179  thrust::host_vector<double> angle = dg::evaluate( dg::coo3, grid.local());
180  for( unsigned i=0; i<pids.size(); i++)
181  {
182  angle[i] += deltaPhi;
183  if( angle[i] >= grid.global().z1()) angle[i] -= grid.global().lz();
184  pids[i] = grid.pidOf( yp[0][i], yp[1][i], angle[i]);
185  if( pids[i] == -1)
186  {
187  std::cerr << "ERROR: PID NOT FOUND!\n";
188  return;
189  }
190  }
191  //construct scatter operation from pids
192  Collective cp( pids, grid.communicator());
193  collP_ = cp;
194  thrust::host_vector<double> pX = collP_.scatter( yp[0]),
195  pY = collP_.scatter( yp[1]),
196  pZ = collP_.scatter( angle);
197  //construt interpolation matrix
198  plus = dg::create::interpolation( pX, pY, pZ, grid.local());
199 
200 
201  //do the same for the minus z-plane
202  for( unsigned i=0; i<pids.size(); i++)
203  {
204  angle[i] -= 2.*deltaPhi;
205  if( angle[i] <= grid.global().z0()) angle[i] += grid.global().lz();
206  pids[i] = grid.pidOf( ym[0][i], ym[1][i], angle[i]);
207  if( pids[i] == -1)
208  {
209  std::cerr << "ERROR: PID NOT FOUND!\n";
210  return;
211  }
212  }
213  Collective cm( pids, grid.communicator());
214  collM_ = cm;
215  pX = collM_.scatter( ym[0]),
216  pY = collM_.scatter( ym[1]),
217  pZ = collM_.scatter( angle);
218  minus = dg::create::interpolation( pX, pY, pZ, grid.local());
219  dg::blas1::axpby( 1., yp[2], 0, hp);
220  dg::blas1::axpby( -1., ym[2], 0, hm);
221  dg::blas1::axpby( 1., hp, +1., hm, hz);
222 
223  interM.resize( collM_.recv_size());
224  interP.resize( collP_.recv_size());

252 MPI_Vector DZ<MPI_Matrix,MPI_Vector>::evaluate( BinaryOp f, unsigned p0)
253 {
254  //let all processes integrate the fieldlines
255  thrust::host_vector<double> global_vec = dz_.evaluate( f, p0);
256  MPI_Vector mpi_vec( g_.n(), g_.Nx(), g_.Ny(), g_.Nz(), g_.communicator());
257  thrust::host_vector<double> vec = mpi_vec.cut_overlap();
258  //now take the relevant part
259  int dims[3], periods[3], coords[3];
260  MPI_Cart_get( g_.communicator(), 3, dims, periods, coords);
261  unsigned Nx = (g_.Nx()-2)*g_.n(), Ny = (g_.Ny()-2)*g_.n(), Nz = g_.Nz();
262  for( unsigned s=0; s<Nz; s++)
263  for( unsigned i=0; i<Ny; i++)
264  for( unsigned j=0; j<Nx; j++)
265  vec[ (s*Ny+i)*Nx + j ]
266  = global_vec[ j + Nx*(coords[0] + dims[0]*( i +Ny*(coords[1] + dims[1]*(s +Nz*coords[2])))) ];
267  mpi_vec.copy_into_interior( vec);
268  return mpi_vec;
269 }

308 void DZ<MPI_Matrix, MPI_Vector>::einsPlus( const MPI_Vector& f, thrust::host_vector<double>& fplus )
309 {
310  const thrust::host_vector<double>& in = f.data();
311  unsigned size = g_.n()*g_.n()*g_.Nx()*g_.Ny();
312  cView fv( in.cbegin(), in.cend());
313 
314  View P( interP.begin(), interP.end() );
315  cusp::multiply( plus, fv, P); //interpolate input vector
316  //gather results from all processes
317  collP_.gather( interP, fplus);
318  //make ghostcells in last plane
319  if( bcz_ != dg::PER && g_.z1() == g_.global().z1())
320  {
321  unsigned i0 = g_.Nz()-1, im = g_.Nz()-2, ip = 0;
322  cView fp( in.cbegin() + ip*size, in.cbegin() + (ip+1)*size);
323  cView f0( in.cbegin() + i0*size, in.cbegin() + (i0+1)*size);
324  cView fm( in.cbegin() + im*size, in.cbegin() + (im+1)*size);
325  View tempPV( fplus.begin() + i0*size, fplus.begin() + (i0+1)*size);
326  View ghostPV( ghostP.begin(), ghostP.end());
327  View ghostMV( ghostM.begin(), ghostM.end());
328  //overwrite tempP
329  cusp::copy( f0, ghostPV);
330  if( bcz_ == dg::DIR || bcz_ == dg::NEU_DIR)
331  {
332  dg::blas1::axpby( 2., right_, -1, ghostP);
333  }
334  if( bcz_ == dg::NEU || bcz_ == dg::DIR_NEU)
335  {
336  dg::blas1::pointwiseDot( right_, hp, ghostM);
337  dg::blas1::axpby( 1., ghostM, 1., ghostP);
338  }
