193 MPI_FieldAligned<LocalMatrix, Communicator, LocalContainer>::MPI_FieldAligned(Field field, const dg::MPI_Grid3d& grid, double eps, Limiter limit, dg::bc globalbcz, double deltaPhi ):
194  hz_( dg::evaluate( dg::zero, grid)), hp_( hz_), hm_( hz_),
195  g_(grid), bcz_(grid.bcz()),
196  dz_(field, grid.global(), eps, limit, globalbcz)
197 {
198  //Resize vector to local 2D grid size
199  dg::Grid2d<double> g2d( g_.x0(), g_.x1(), g_.y0(), g_.y1(), g_.n(), g_.Nx(), g_.Ny());
200  unsigned size = g2d.size();
201  limiter_ = dg::evaluate( limit, g2d);
202  right_ = left_ = dg::evaluate( zero, g2d);
203  ghostM.resize( size); ghostP.resize( size);
204  //set up grid points as start for fieldline integrations
205  std::vector<thrust::host_vector<double> > y( 3);
206  y[0] = dg::evaluate( dg::coo1, grid.local());
207  y[1] = dg::evaluate( dg::coo2, grid.local());
208  y[2] = dg::evaluate( dg::zero, grid.local());//distance (not angle)
209  //integrate to next z-planes
210  std::vector<thrust::host_vector<double> > yp(y), ym(y);
211  if(deltaPhi<=0) deltaPhi = g_.hz();
212  else assert( g_.Nz() == 1 || grid.hz()==deltaPhi);
213  unsigned localsize = grid.local().size();
214 #ifdef _OPENMP
215 #pragma omp parallel for firstprivate(field)
216 #endif //_OPENMP
217  for( unsigned i=0; i<localsize; i++)
218  {
219  thrust::host_vector<double> coords(3), coordsP(3), coordsM(3);
220  coords[0] = y[0][i], coords[1] = y[1][i], coords[2] = y[2][i];
221  double phi1 = deltaPhi;
222  boxintegrator( field, g_.global(), coords, coordsP, phi1, eps, globalbcz);
223  phi1 = -deltaPhi;
224  boxintegrator( field, g_.global(), coords, coordsM, phi1, eps, globalbcz);
225  yp[0][i] = coordsP[0], yp[1][i] = coordsP[1], yp[2][i] = coordsP[2];
226  ym[0][i] = coordsM[0], ym[1][i] = coordsM[1], ym[2][i] = coordsM[2];
227  }
228 
229 
230  //determine pid of result
231  thrust::host_vector<int> pids( localsize);
232  thrust::host_vector<double> angle = dg::evaluate( dg::coo3, grid.local());
233  for( unsigned i=0; i<pids.size(); i++)
234  {
235  angle[i] += deltaPhi;
236  if( angle[i] >= grid.global().z1()) angle[i] -= grid.global().lz();
237  pids[i] = grid.pidOf( yp[0][i], yp[1][i], angle[i]);
238  if( pids[i] == -1)
239  {
240  std::cerr << "ERROR: PID NOT FOUND!\n";
241  return;
242  }
243  }
244  //construct scatter operation from pids
245  Communicator cp( pids, grid.communicator());
246  thrust::host_vector<double> pX = cp.collect( yp[0]),
247  pY = cp.collect( yp[1]),
248  pZ = cp.collect( angle);
249  //construt interpolation matrix
250  LocalMatrix inter = dg::create::interpolation( pX, pY, pZ, grid.local());
251  plus = ColDistMat<LocalMatrix, Communicator>(inter, cp);
252  LocalMatrix interT;
253  cusp::transpose( inter, interT);
254  plusT = RowDistMat<LocalMatrix, Communicator>( interT, cp);
255 
256 
257 
258  //do the same for the minus z-plane
259  for( unsigned i=0; i<pids.size(); i++)
260  {
261  angle[i] -= 2.*deltaPhi;
262  if( angle[i] <= grid.global().z0()) angle[i] += grid.global().lz();
263  pids[i] = grid.pidOf( ym[0][i], ym[1][i], angle[i]);
264  if( pids[i] == -1)
265  {
266  std::cerr << "ERROR: PID NOT FOUND!\n";
267  return;
268  }
269  }
270  Communicator cm( pids, grid.communicator());
271  pX = cm.collect( ym[0]),
272  pY = cm.collect( ym[1]),
273  pZ = cm.collect( angle);
274  inter = dg::create::interpolation( pX, pY, pZ, grid.local());
275  minus = ColDistMat<LocalMatrix, Communicator>(inter, cm);
276  cusp::transpose( inter, interT);
277  minusT = RowDistMat<LocalMatrix, Communicator>( interT, cm);
278  //copy to device
279  thrust::copy( yp[2].begin(), yp[2].end(), hp_.data().begin());
280  thrust::copy( ym[2].begin(), ym[2].end(), hm_.data().begin());
281  dg::blas1::scal( hm_, -1.);
282  dg::blas1::axpby( 1., hp_, +1., hm_, hz_);
283 }

287 MPI_Vector<container> MPI_FieldAligned<M,C,container>::evaluate( BinaryOp f, unsigned p0) const
288 {
289  container global_vec = dz_.evaluate( f, p0);
290  container vec = dg::evaluate( dg::zero, g_.local());
291  //now take the relevant part
292  int dims[3], periods[3], coords[3];
293  MPI_Cart_get( g_.communicator(), 3, dims, periods, coords);
294  unsigned Nx = g_.Nx()*g_.n(), Ny = g_.Ny()*g_.n(), Nz = g_.Nz();
295  for( unsigned s=0; s<Nz; s++)
296  for( unsigned i=0; i<Ny; i++)
297  for( unsigned j=0; j<Nx; j++)
298  vec[ (s*Ny+i)*Nx + j ]
299  = global_vec[ j + Nx*(coords[0] + dims[0]*( i +Ny*(coords[1] + dims[1]*(s +Nz*coords[2])))) ];
300 
301  return MPI_Vector<container>( vec, g_.communicator());
302 }

324 void MPI_FieldAligned<M,C,container>::einsPlus( const MPI_Vector<container>& f, MPI_Vector<container>& fplus )
325 {
326  dg::blas2::detail::doSymv( plus, f, fplus, MPIMatrixTag(), MPIVectorTag(), MPIVectorTag());
327  //make ghostcells in last plane
328  const container& in = f.data();
329  container& out = fplus.data();
330  unsigned size = g_.n()*g_.n()*g_.Nx()*g_.Ny();
331  if( bcz_ != dg::PER && g_.z1() == g_.global().z1())
332  {
333  unsigned i0 = g_.Nz()-1, im = g_.Nz()-2, ip = 0;
334  cView fp( in.cbegin() + ip*size, in.cbegin() + (ip+1)*size);
335  cView f0( in.cbegin() + i0*size, in.cbegin() + (i0+1)*size);
336  cView fm( in.cbegin() + im*size, in.cbegin() + (im+1)*size);
337  View outV( out.begin() + i0*size, out.begin() + (i0+1)*size);
338  View ghostPV( ghostP.begin(), ghostP.end());
339  View ghostMV( ghostM.begin(), ghostM.end());
340  //overwrite out
341  cusp::copy( f0, ghostPV);
342  if( bcz_ == dg::DIR || bcz_ == dg::NEU_DIR)
343  {
344  dg::blas1::axpby( 2., right_, -1, ghostP);
345  }
346  if( bcz_ == dg::NEU || bcz_ == dg::DIR_NEU)
347  {
348  //note that hp_ is 3d and the rest 2d
349  thrust::transform( right_.begin(), right_.end(), hp_.data().begin(), ghostM.begin(), thrust::multiplies<double>());
350  dg::blas1::axpby( 1., ghostM, 1., ghostP);
351  }
352  cusp::blas::axpby( ghostPV, outV, ghostPV, 1.,-1.);
353  dg::blas1::pointwiseDot( limiter_, ghostP, ghostP);
354  cusp::blas::axpby( ghostPV, outV, outV, 1.,1.);
355  }
