275  int remain_dims[] = {true,true,false}; //true true false
276  MPI_Cart_sub( grid.communicator(), remain_dims, &planeComm);

278  unsigned localsize = g2d.size();
279  limiter_ = dg::evaluate( limit, g2d.local());
280  right_ = left_ = dg::evaluate( zero, g2d.local());
281  ghostM.resize( localsize); ghostP.resize( localsize);
282  //set up grid points as start for fieldline integrations
283  std::vector<thrust::host_vector<double> > y( 3);
284  y[0] = dg::evaluate( dg::coo1, g2d.local());
285  y[1] = dg::evaluate( dg::coo2, g2d.local());
286  y[2] = dg::evaluate( dg::zero, g2d.local());//distance (not angle)
287  //integrate to next z-planes
288  std::vector<thrust::host_vector<double> > yp(y), ym(y);
289  if(deltaPhi<=0) deltaPhi = g_.hz();
290  else assert( g_.Nz() == 1 || grid.hz()==deltaPhi);
291 #ifdef _OPENMP
292 #pragma omp parallel for shared(field)

295  {
296  thrust::host_vector<double> coords(3), coordsP(3), coordsM(3);
297  coords[0] = y[0][i], coords[1] = y[1][i], coords[2] = y[2][i];

334  return;
335  }

340  pY = cm.collect( ym[1]);
341  minus = dg::create::interpolation( pX, pY, g2d.local(), globalbcz); //inner points hopefully never lie exactly on local boundary
342  cusp::transpose( minus, minusT);
343  //copy to device
344  for( unsigned i=0; i<g_.Nz(); i++)

348  }

351  for( unsigned i=0; i<g_.Nz(); i++)
352  {
353  tempXYplus_[i].resize( commXYplus_.size());
354  tempXYminus_[i].resize( commXYminus_.size());
355  temp_[i].resize( localsize);
356  }

358  tempZ_.resize( commZ_.size());

363 MPI_Vector<container> MPI_FieldAligned<M,C,container>::evaluate( BinaryOp f, unsigned p0) const
364 {
365  container global_vec = dz_.evaluate( f, p0);
366  container vec = dg::evaluate( dg::zero, g_.local());
367  //now take the relevant part
368  int dims[3], periods[3], coords[3];
369  MPI_Cart_get( g_.communicator(), 3, dims, periods, coords);
370  unsigned Nx = g_.Nx()*g_.n(), Ny = g_.Ny()*g_.n(), Nz = g_.Nz();
371  for( unsigned s=0; s<Nz; s++)
372  for( unsigned i=0; i<Ny; i++)
373  for( unsigned j=0; j<Nx; j++)
374  vec[ (s*Ny+i)*Nx + j ]
375  = global_vec[ j + Nx*(coords[0] + dims[0]*( i +Ny*(coords[1] + dims[1]*(s +Nz*coords[2])))) ];
376 
377  return MPI_Vector<container>( vec, g_.communicator());
378 }

405  int size2d = g_.n()*g_.n()*g_.Nx()*g_.Ny();
406  int dims[3], periods[3], coords[3];
407  MPI_Cart_get( g_.communicator(), 3, dims, periods, coords);
408  int sizeXY = dims[0]*dims[1];
409  int sizeZ = dims[2];
410 
411  //1. compute 2d interpolation in every plane and store in temp_
412  if( sizeXY != 1) //communication needed
413  {
414  for( int i0=0; i0<(int)g_.Nz(); i0++)
415  {
416  cView inV( in.cbegin() + i0*plus.num_cols, in.cbegin() + (i0+1)*plus.num_cols);
417  View tempV( tempXYplus_[i0].begin(), tempXYplus_[i0].end() );
418  cusp::multiply( plus, inV, tempV);
419  //exchange data in XY
420  commXYplus_.send_and_reduce( tempXYplus_[i0], temp_[i0]);

422  }

424  {
425  for( int i0=0; i0<(int)g_.Nz(); i0++)
426  {
427  cView inV( in.cbegin() + i0*plus.num_cols, in.cbegin() + (i0+1)*plus.num_cols);
428  View tempV( temp_[i0].begin(), temp_[i0].end() );

430  }
431  }
432  //2. reorder results and communicate halo in z

434  {
435  int ip = i0 + 1;
436  if( ip > (int)g_.Nz()-1) ip -= (int)g_.Nz();

443  }
444

447  if( bcz_ != dg::PER && g_.z1() == g_.global().z1())
448  {
449  unsigned i0 = g_.Nz()-1, im = g_.Nz()-2, ip = 0;
450  cView fp( in.cbegin() + ip*size, in.cbegin() + (ip+1)*size);
451  cView f0( in.cbegin() + i0*size, in.cbegin() + (i0+1)*size);
452  cView fm( in.cbegin() + im*size, in.cbegin() + (im+1)*size);
453  View outV( out.begin() + i0*size, out.begin() + (i0+1)*size);
454  View ghostPV( ghostP.begin(), ghostP.end());
455  View ghostMV( ghostM.begin(), ghostM.end());
456  //overwrite out
457  cusp::copy( f0, ghostPV);
458  if( bcz_ == dg::DIR || bcz_ == dg::NEU_DIR)
459  {
460  dg::blas1::axpby( 2., right_, -1, ghostP);
461  }
462  if( bcz_ == dg::NEU || bcz_ == dg::DIR_NEU)
463  {
464  //note that hp_ is 3d and the rest 2d
465  thrust::transform( right_.begin(), right_.end(), hp_.data().begin(), ghostM.begin(), thrust::multiplies<double>());
466  dg::blas1::axpby( 1., ghostM, 1., ghostP);
467  }
468  cusp::blas::axpby( ghostPV, outV, ghostPV, 1.,-1.);
469  dg::blas1::pointwiseDot( limiter_, ghostP, ghostP);
470  cusp::blas::axpby( ghostPV, outV, outV, 1.,1.);
471  }
