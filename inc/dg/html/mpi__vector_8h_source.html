108 void MPI_Vector::x_col( MPI_Comm comm)
109 {
110  //shift data in zero-th dimension
111  MPI_Status status;
112  int n = n_;
113  int cols = Nx_;
114  int rows = n_*Ny_*Nz_;
115  //create buffer before sending single cells (1 is left side, 2 is right side)
116  thrust::host_vector<double> sendbuffer1( rows*n, 0);
117  thrust::host_vector<double> recvbuffer1( rows*n, 0);
118  thrust::host_vector<double> sendbuffer2( rows*n, 0);
119  thrust::host_vector<double> recvbuffer2( rows*n, 0);
120  //copy into buffers
121  for( int i=0; i<rows; i++)
122  for( int j=0; j<n; j++)
123  {
124  sendbuffer1[i*n+j] = data_[(i*cols + 1 )*n+j];
125  sendbuffer2[i*n+j] = data_[(i*cols + cols - 2)*n+j];
126  }
127  int source, dest;
128  MPI_Cart_shift( comm, 0, -1, &source, &dest);
129  MPI_Sendrecv( sendbuffer1.data(), rows*n, MPI_DOUBLE, //sender
130  dest, 3, //destination
131  recvbuffer2.data(), rows*n, MPI_DOUBLE, //receiver
132  source, 3, //source
133  MPI_COMM_WORLD, &status);
134  MPI_Cart_shift( comm, 0, +1, &source, &dest);
135  MPI_Sendrecv( sendbuffer2.data(), rows*n, MPI_DOUBLE, //sender
136  dest, 9, //destination
137  recvbuffer1.data(), rows*n, MPI_DOUBLE, //receiver
138  source, 9, //source
139  MPI_COMM_WORLD, &status);
140  //copy back into vector
141  for( int i=0; i<rows; i++)
142  for( int j=0; j<n; j++)
143  {
144  data_[(i*cols )*n+j] = recvbuffer1[i*n+j];
145  data_[(i*cols + cols - 1)*n+j] = recvbuffer2[i*n+j];
146  }
