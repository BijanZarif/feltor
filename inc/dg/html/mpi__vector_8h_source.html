231 template<class I, class V>
232 void NearestNeighborComm<I,V>::construct( int n, const int dimensions[3], MPI_Comm comm, int direction)
233 {
234  silent_=false;
235  int ndims;
236  MPI_Cartdim_get( comm, &ndims);
237  int dims[ndims], periods[ndims], coords[ndims];
238  MPI_Cart_get( comm, ndims, dims, periods, coords);
239  if( dims[direction] == 1) silent_ = true;
240  n_=n;
241  dim_[0] = dimensions[0], dim_[1] = dimensions[1], dim_[2] = dimensions[2];
242  comm_ = comm;
243  direction_ = direction;
244  assert( 0<=direction);
245  assert( direction <3);
246  thrust::host_vector<int> hbgather1(buffer_size()), hbgather2(hbgather1), hbscattr1(buffer_size()), hbscattr2(hbscattr1);
247  switch( direction)
248  {
249  case( 0):
250  for( int i=0; i<dim_[2]*dim_[1]; i++)
251  {
252  for( int j=0; j<n_; j++)
253  {
254  hbgather1[i*n+j] = (i*dim_[0] + j);
255  hbgather2[i*n+j] = (i*dim_[0] + dim_[0] - n + j);
256  hbscattr1[i*n+j] = (i*(2*n) + j);
257  hbscattr2[i*n+j] = (i*(2*n)+ (2*n) - n + j);
258  }
259  }
260  break;
261  case( 1):
262  for( int i=0; i<dim_[2]; i++)
263  {
264  for( int j=0; j<n; j++)
265  for( int k=0; k<dim_[0]; k++)
266  {
267  hbgather1[(i*n+j)*dim_[0]+k] =
268  (i*dim_[1] + j)*dim_[0] + k;
269  hbgather2[(i*n+j)*dim_[0]+k] =
270  (i*dim_[1] + dim_[1] - n + j)*dim_[0] + k;
271  hbscattr1[(i*n+j)*dim_[0]+k] =
272  (i*( 2*n) + j)*dim_[0] + k;
273  hbscattr2[(i*n+j)*dim_[0]+k] =
274  (i*( 2*n) + ( 2*n) - n + j)*dim_[0] + k;
275  }
276  }
277  break;
278  case( 2):
279  for( int i=0; i<n; i++)
280  {
281  for( int j=0; j<dim_[0]*dim_[1]; j++)
282  {
283  hbgather1[i*dim_[0]*dim_[1]+j] = i*dim_[0]*dim_[1] + j;
284  hbgather2[i*dim_[0]*dim_[1]+j] = (i+dim_[2]-n)*dim_[0]*dim_[1] + j;
285  hbscattr1[i*dim_[0]*dim_[1]+j] = i*dim_[0]*dim_[1] + j;
286  hbscattr2[i*dim_[0]*dim_[1]+j] = (i+( 2*n)-n)*dim_[0]*dim_[1] + j;
287  }
288  }
289  break;
290  }
291  buffer_gather1 =hbgather1, buffer_gather2 =hbgather2;
292  buffer_scatter1=hbscattr1, buffer_scatter2=hbscattr2;
293 }
