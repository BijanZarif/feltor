614 int MPI_Grid2d::pidOf( double x, double y) const
615 {
616  int dims[2], periods[2], coords[2];
617  MPI_Cart_get( comm, 2, dims, periods, coords);
618  coords[0] = (unsigned)floor( (x-g.x0())/g.lx()*(double)dims[0] );
619  coords[1] = (unsigned)floor( (y-g.y0())/g.ly()*(double)dims[1] );
620  //if point lies on or over boundary of last cell shift into current cell (not so good for periodic boundaries)
621  coords[0]=(coords[0]==dims[0]) ? coords[0]-1 :coords[0];
622  coords[1]=(coords[1]==dims[1]) ? coords[1]-1 :coords[1];
623  int rank;
624  if( MPI_Cart_rank( comm, coords, &rank) == MPI_SUCCESS )
625  return rank;
626  else
627  return -1;
628 }
