442 int MPI_Grid3d::pidOf( double x, double y, double z) const
443 {
444  int dims[3], periods[3], coords[3];
445  MPI_Cart_get( comm, 3, dims, periods, coords);
446  //points in the (outer) ghost cells layer? (note the global grid used)
447  if( x < g.x0() && x >= g.x0() - g.hx()) x += g.hx();
448  if( x > g.x1() && x <= g.x1() + g.hx()) x -= g.hx();
449  if( y < g.y0() && y >= g.y0() - g.hy()) y += g.hy();
450  if( y > g.y1() && y <= g.y1() + g.hy()) y -= g.hy();
451  coords[0] = (unsigned)floor( (x-g.x0())/g.lx()*(double)dims[0] );
452  coords[1] = (unsigned)floor( (y-g.y0())/g.ly()*(double)dims[1] );
453  coords[2] = (unsigned)floor( (z-g.z0())/g.lz()*(double)dims[2] );
454  //if point lies on boundary of last cell ... (not so good for periodic boundaries)
455  coords[0]=(coords[0]==dims[0]) ? coords[0]-1 :coords[0];
456  coords[1]=(coords[1]==dims[1]) ? coords[1]-1 :coords[1];
457  coords[2]=(coords[2]==dims[2]) ? coords[2]-1 :coords[2];
458  int rank;
459  if( MPI_Cart_rank( comm, coords, &rank) == MPI_SUCCESS )
460  return rank;
461  else
462  return -1;
463 }
