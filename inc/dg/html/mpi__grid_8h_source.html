613 int MPI_Grid2d::pidOf( double x, double y) const
614 {
615  int dims[2], periods[2], coords[2];
616  MPI_Cart_get( comm, 2, dims, periods, coords);
617  coords[0] = (unsigned)floor( (x-g.x0())/g.lx()*(double)dims[0] );
618  coords[1] = (unsigned)floor( (y-g.y0())/g.ly()*(double)dims[1] );
619  //if point lies on or over boundary of last cell shift into current cell (not so good for periodic boundaries)
620  coords[0]=(coords[0]==dims[0]) ? coords[0]-1 :coords[0];
621  coords[1]=(coords[1]==dims[1]) ? coords[1]-1 :coords[1];
622  int rank;
623  if( MPI_Cart_rank( comm, coords, &rank) == MPI_SUCCESS )
624  return rank;
625  else
626  return -1;
627 }
