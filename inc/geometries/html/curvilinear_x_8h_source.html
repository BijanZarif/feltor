412  //compute metric elements from xr, xz, yr, yz, r and z
413  void construct_metric()
414  {
415  //std::cout << "CONSTRUCTING METRIC\n";
416  thrust::host_vector<double> tempxx( r_), tempxy(r_), tempyy(r_), tempvol(r_);
417  for( unsigned idx=0; idx<this->size(); idx++)
418  {
419  tempxx[idx] = (xr_[idx]*xr_[idx]+xz_[idx]*xz_[idx]);
420  tempxy[idx] = (yr_[idx]*xr_[idx]+yz_[idx]*xz_[idx]);
421  tempyy[idx] = (yr_[idx]*yr_[idx]+yz_[idx]*yz_[idx]);
422  tempvol[idx] = r_[idx]/sqrt(tempxx[idx]*tempyy[idx]-tempxy[idx]*tempxy[idx]);
423  }
424  g_xx_=tempxx, g_xy_=tempxy, g_yy_=tempyy, vol_=tempvol;

427  thrust::host_vector<double> ones = dg::evaluate( dg::one, *this);
428  dg::blas1::pointwiseDivide( ones, r_, tempxx);
429  dg::blas1::pointwiseDivide( tempxx, r_, tempxx); //1/R^2
430  g_pp_=tempxx;
431  }
