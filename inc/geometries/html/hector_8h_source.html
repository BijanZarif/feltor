90 void compute_zev(
91  const thrust::host_vector<double>& etaV,
92  const thrust::host_vector<double>& v_vec,
93  thrust::host_vector<double>& eta,
94  const dg::Grid2d& g2d
95  )
96 {
97  Interpolate iter( thrust::host_vector<double>( etaV.size(), 0), etaV, g2d);
98  eta.resize( v_vec.size());
99  thrust::host_vector<double> eta_old(v_vec.size(), 0), eta_diff( eta_old);
100  thrust::host_vector<double> begin( 2, 0), end(begin), temp(begin);
101  begin[0] = 0., begin[1] = 0.;
102  unsigned steps = 1;
103  double eps = 1e10, eps_old=2e10;
104  while( (eps < eps_old||eps > 1e-7) && eps > 1e-12)
105  {
106  //begin is left const
107  eps_old = eps, eta_old = eta;
108  dg::stepperRK17( iter, begin, end, 0, v_vec[0], steps);
109  eta[0] = end[1];
110  for( unsigned i=1; i<v_vec.size(); i++)
111  {
112  temp = end;
113  dg::stepperRK17( iter, temp, end, v_vec[i-1], v_vec[i], steps);
114  eta[i] = end[1];
115  }
116  temp = end;
117  dg::stepperRK17( iter, temp, end, v_vec[v_vec.size()-1], 2.*M_PI, steps);
118  dg::blas1::axpby( 1., eta, -1., eta_old, eta_diff);
119  eps = sqrt( dg::blas1::dot( eta_diff, eta_diff) / dg::blas1::dot( eta, eta));
120  //std::cout << "rel. error is "<<eps<<" with "<<steps<<" steps\n";
121  //std::cout << "abs. error is "<<( 2.*M_PI-end[1])<<"\n";
122  steps*=2;
123  }

126 void construct_grid(
127  const thrust::host_vector<double>& zetaU, //2d Zeta component
128  const thrust::host_vector<double>& etaU, //2d Eta component
129  const thrust::host_vector<double>& u_vec, //1d u values
130  const thrust::host_vector<double>& zeta_init, //1d intial values
131  const thrust::host_vector<double>& eta_init, //1d intial values
132  thrust::host_vector<double>& zeta,
133  thrust::host_vector<double>& eta,
134  const dg::Grid2d& g2d
135  )
136 {
137  Interpolate inter( zetaU, etaU, g2d);
138  unsigned N = 1;
139  double eps = 1e10, eps_old=2e10;
140  std::vector<thrust::host_vector<double> > begin(2);
141  begin[0] = zeta_init, begin[1] = eta_init;
142  //now we have the starting values
143  std::vector<thrust::host_vector<double> > end(begin), temp(begin);
144  unsigned sizeU = u_vec.size(), sizeV = zeta_init.size();
145  unsigned size2d = sizeU*sizeV;
146  zeta.resize(size2d), eta.resize(size2d);
147  double u0=0, u1 = u_vec[0];
148  thrust::host_vector<double> zeta_old(zeta), zeta_diff( zeta), eta_old(eta), eta_diff(eta);
149  while( (eps < eps_old || eps > 1e-6) && eps > 1e-7)
150  {
151  zeta_old = zeta, eta_old = eta; eps_old = eps;
152  temp = begin;
154  for( unsigned i=0; i<sizeU; i++)
155  {
156  u0 = i==0?0:u_vec[i-1], u1 = u_vec[i];
157  dg::stepperRK17( inter, temp, end, u0, u1, N);
158  for( unsigned j=0; j<sizeV; j++)
159  {
160  unsigned idx = j*sizeU+i;
161  zeta[idx] = end[0][j], eta[idx] = end[1][j];
162  }
163  temp = end;
164  }
165  dg::blas1::axpby( 1., zeta, -1., zeta_old, zeta_diff);
166  dg::blas1::axpby( 1., eta, -1., eta_old, eta_diff);
167  dg::blas1::pointwiseDot( zeta_diff, zeta_diff, zeta_diff);
168  dg::blas1::pointwiseDot( 1., eta_diff, eta_diff, 1., zeta_diff);
169  eps = sqrt( dg::blas1::dot( zeta_diff, zeta_diff)/sizeU/sizeV);
170  //std::cout << "Effective Absolute diff error is "<<eps<<" with "<<N<<" steps\n";
171  N*=2;
172  }

495  void construct(const container& u, double psi0, double psi1, Chi_XX chi_XX, Chi_XY chi_XY, Chi_YY chi_YY)
496  {
497  //now compute u_zeta and u_eta
498  Matrix dzeta = dg::create::dx( g2d_, dg::DIR);
499  Matrix deta = dg::create::dy( g2d_, dg::PER);
500  container u_zeta(u), u_eta(u);
501  dg::blas2::symv( dzeta, u, u_zeta);
502  dg::blas1::plus( u_zeta, (psi1-psi0)/g2d_.lx());
503  dg::blas2::symv( deta, u, u_eta);
504 
505 
506  thrust::host_vector<double> chi_ZZ, chi_ZE, chi_EE;
507  dg::geo::pushForwardPerp( chi_XX, chi_XY, chi_YY, chi_ZZ, chi_ZE, chi_EE, g2d_);
508  container chiZZ, chiZE, chiEE;
509  dg::blas1::transfer( chi_ZZ, chiZZ);
510  dg::blas1::transfer( chi_ZE, chiZE);
511  dg::blas1::transfer( chi_EE, chiEE);
512 
513  //now compute ZetaU and EtaU
514  container temp_zeta(u), temp_eta(u);
515  dg::blas1::pointwiseDot( chiZZ, u_zeta, temp_zeta);
516  dg::blas1::pointwiseDot( 1. ,chiZE, u_eta, 1., temp_zeta);
517  dg::blas1::pointwiseDot( chiZE, u_zeta, temp_eta);
518  dg::blas1::pointwiseDot( 1. ,chiEE, u_eta, 1., temp_eta);
519  container temp_scalar(u);
520  dg::blas1::pointwiseDot( u_zeta, temp_zeta, temp_scalar);
521  dg::blas1::pointwiseDot( 1., u_eta, temp_eta, 1., temp_scalar);
522  container zetaU=temp_zeta, etaU=temp_eta;
523  dg::blas1::pointwiseDivide( zetaU, temp_scalar, zetaU);
524  dg::blas1::pointwiseDivide( etaU, temp_scalar, etaU);
525  //now compute etaV and its inverse
526  container etaVinv(u_zeta), etaV(etaVinv);
527  dg::blas1::pointwiseDot( etaVinv, chiZZ, etaVinv);
528  dg::geo::multiplyPerpVolume( etaVinv, g2d_);
529  dg::blas1::transform( etaVinv, etaV, dg::INVERT<double>());
530  thrust::host_vector<double> etaVinv_h;
531  dg::blas1::transfer( etaVinv, etaVinv_h);
532  //now compute v_zeta and v_eta
533  container v_zeta(u), v_eta(u);
534  dg::blas1::axpby( -1., temp_eta, 0.,v_zeta);
535  dg::blas1::axpby( +1., temp_zeta, 0.,v_eta);
536  dg::geo::multiplyPerpVolume( v_zeta, g2d_);
537  dg::geo::multiplyPerpVolume( v_eta, g2d_);
538 
539  //construct c0 and scale all vector components with it
540  c0_ = fabs( detail::construct_c0( etaVinv_h, g2d_));
541  if( psi1 < psi0) c0_*=-1;
542  lu_ = c0_*(psi1-psi0);
543  //std::cout << "c0 is "<<c0_<<"\n";
544  dg::blas1::scal( etaV, 1./c0_);
545  dg::blas1::scal( zetaU, 1./c0_);
546  dg::blas1::scal( etaU, 1./c0_);
547  dg::blas1::scal( u_zeta, c0_);
548  dg::blas1::scal( v_zeta, c0_);
549  dg::blas1::scal( u_eta, c0_);
550  dg::blas1::scal( v_eta, c0_);
551  //transfer to host
552  detail::transform( u_zeta, u_eta, ux_, uy_, g2d_);
553  detail::transform( v_zeta, v_eta, vx_, vy_, g2d_);
554  dg::blas1::transfer( etaV, etaV_);
555  dg::blas1::transfer( etaU, etaU_);
556  dg::blas1::transfer( zetaU, zetaU_);
557  }
